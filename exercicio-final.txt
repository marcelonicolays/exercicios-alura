// Exercicio 1

function verificarPositivo(num){
    if (num > 0) {
	alert('Número positivo');
    } else if (num = 0){
	alert('Número é zero');
    } else{
	alert('Número negativo');
    }
}

    if (ano % 4 == 0 && ano % 100 != 0){
	alert ('Ano é Bissexto!');
    } else if (ano % 400 == 0){
	alert ('Ano é Bissexto!');
    }else {
	alert ('Ano não é Bissexto!');
    }
}

function media(num1, num2){
    if (isNaN(num1) || isNaN(num2)){
	alert('Digite númeors válidos');
	return;
}
    let resultado = (num1 + num2) / 2;
    alert(`Média dos números ${num1} e ${num2} é ${resultado}!`);
}

// Exercicio 11

function calculos(numeros){
	let numerosPares = [];
	let numerosImpares = [];
	let soma = 0;
	let produto = 1;
	for (let i = 0; i < numeros.length; i++){
	   if (numeros[i] % 2 === 0){
		numerosPares.push(numeros[i]);
	   } else {
		numerosImpares.push(numeros[i]);
	   }
	}
	for (let i = 0; i < numerosPares.length; i++){
	   soma += numerosPares[i];
	}
	for (let i = 0; i < numerosImpares.length; i++){
	   produto *= numerosImpares[i];
	}
	alert(`A soma dos números pares é ${soma} e o produto dos númeors ímpares é ${produto}`);
}
