// Exercicio 1

function verificarPositivo(num){
    if (num > 0) {
	return alert('Número positivo');
    } else if (num = 0){
	return alert('Número é zero');
    } else{
	return alert('Número negativo');
    }
}

// Exercicio 2

function maioridade(idade){
    if (idade >= 18){
	return alert('É maior de idade');
    } else {
	return alert('É menor de idade');
    }
}

// Exercicio 3

function validaStr(string){
    if (string = ''){
	return alert('String vazia');
    } else {
	return alert('String possui valor');
    }
}

// Exercicio 4

function anoBissexto(ano){
    if (ano % 4 == 0 && ano % 100 != 0){
	alert ('Ano é Bissexto!');
    } else if (ano % 400 == 0){
	alert ('Ano é Bissexto!');
    }else {
	alert ('Ano não é Bissexto!');
    }
}
